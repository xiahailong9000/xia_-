夏海龙方法


using UnityEngine;
using System.Collections;
using System.Collections.Generic;
using System.Linq;

public  class xia : MonoBehaviour {
 void Start () {}
 void Update () {}
    #region 夏海龙的方法 --------------------------------------
 
    public static string[] S_添加字符串(string[] list, string key){
        List<string> l = list.ToList();
        l.Add(key);
       return l.ToArray();
    }//向数组中添加字符串
    public static string[] S_删除字符串(string[] list, string key) {
        List<string> l = new List<string>();
        foreach (string s in list) {
            if (s != key)l.Add(s);
        } return l.ToArray();
    }//向数组中删除字符串
    public static GameObject[] S_添加模型(GameObject[] list, GameObject key){
        List<GameObject> l = list.ToList();
        l.Add(key);
        return l.ToArray();
    }//向数组中添加物体
    public static GameObject[] S_删除模型(GameObject[] list, GameObject key){
        List<GameObject> l = new List<GameObject>();
        foreach (GameObject s in list) {
            if (s != key) l.Add(s);
        } return l.ToArray();
    }//向数组中删除物体
    public static GameObject[] S_获取所有子物体(GameObject oo){
        Transform[] tts = oo.GetComponentsInChildren<Transform>(true); //获取子物体的所有变换
        GameObject[] oos = new GameObject[tts.Length];
        for (int i = 0; i < tts.Length; i++){
            oos[i] = tts[i].gameObject;
        }
        oos = S_删除模型(oos, oo);
        return oos;
    }  //获取所有子物体
    public static bool S_字符串是否存在(string[] array, string s){
        List<string> list = new List<string>(array);
        if (list.Contains(s)){
            return true;
        }else{
            return false;
        }
    }
    #endregion
}
