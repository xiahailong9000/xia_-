using UnityEngine;
using System.Collections;

public class C_掉血 : MonoBehaviour {
    public Vector3 o_初始位置, o_重力, o_抛力;
    public GameObject gg;
    public float  o_比例;
 void Start () {
        if (gameObject.activeSelf == true){
            gg = new GameObject();
            gg.transform.parent = gameObject.transform.parent;
            Destroy(gg, o_比例);
            Destroy(gameObject, o_比例);
            o_重力 = new Vector3(0, -4 * o_比例, 0);
            o_抛力 = new Vector3(Random.Range(-1.7f * o_比例, 1.7f * o_比例), Random.Range(2f * o_比例, 3f * o_比例), Random.Range(-1.7f * o_比例, 1.7f * o_比例));
        }
 }
 void Update () {
        if (gameObject.activeSelf == true){
           
            o_抛力 += o_重力 * Time.deltaTime;
            o_初始位置 += o_抛力 * Time.deltaTime;
            gg.transform.position = o_初始位置;
            Vector3 tt2 = Camera.main.WorldToScreenPoint(gg.transform.position);
         //   transform.position = new Vector3(tt2.x, Screen.height - tt2.y, 0);
            transform.position = new Vector3(tt2.x,tt2.y, 0);
        }
 }
}
