--面向对象----------------------------------------------------------------
function S_克隆表(tab)
    local tt={}
    for kk,vv in pairs(tab) do
        tt[kk]=vv
    end
    return tt
end

------------------------方法闭包类----------------------------------------
function C_mm()
    local mm={}
    local function S_初始化()
         mm.name="C_mm___dddddddddd";
         print(mm.name);
    end
    mm.S_002=function()
      print("我是方法002")
    end
    S_初始化();
    return mm;
end
--------------------------类的继承------------------------------------
