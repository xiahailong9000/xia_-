internal class C_计时器{
    [DllImport("Kernel32.dll")]
    private static extern bool QueryPerformanceCounter(out long lpPerformanceCount);
    [DllImport("Kernel32.dll")]
    private static extern bool QueryPerformanceFrequency(out long lpFrequency);
    long o_开始时间, o_停止时间;
    long o_频率;
    public C_计时器(){
        o_开始时间 = 0;
        o_停止时间 = 0;
        if (QueryPerformanceFrequency(out o_频率) == false){
   throw new Win32Exception();  //不支持高性能计数器
        }
  S_开始计时();
    }
 public double S_结果 {//返回计时器的持续时间（以毫秒为单位）   
  get {
   S_停止();
   return (double)(o_停止时间-o_开始时间)*1000/(double)o_频率;
  }
 }
 void S_开始计时() { //启动定时器
  Thread.Sleep(0);  //让我们做等待的线程有工作
        QueryPerformanceCounter(out o_开始时间);
    }
 void S_停止() {// 停止计时器
        QueryPerformanceCounter(out o_停止时间);
    }
}
