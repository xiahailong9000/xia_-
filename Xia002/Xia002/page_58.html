using UnityEngine;
using System.Collections;
using System.Reflection;
using System;
public class C_002man : MonoBehaviour {
 void Start () {
        GameObject gg =  GameObject.Find("物体");
        Component cc = gg.GetComponent("脚本");
        Type type =cc.GetType();
        PropertyInfo ff = type.GetProperty("ii"); //获取指定名称的属性
        int ii = (int)ff.GetValue(cc, null); //获取属性值
        print(ii + "   ddd");
        MethodInfo mm = type.GetMethod("S_002");//获取当前C_ff类的S_002方法
        string ss = mm.Invoke(cc, null).ToString();   //调用对象cc的S_002方法  S_002无参数
        print(ss);
 }
}
