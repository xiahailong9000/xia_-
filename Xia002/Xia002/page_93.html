特效_序列图


using UnityEngine;
using System.Collections;
public class C_序列图 : MonoBehaviour {
    public int o_x,o_y;
    MeshRenderer o_mm;
    float offset;
 void Start () {
        o_mm=gameObject.GetComponent<MeshRenderer>();
        o_mm.material.SetTextureScale("_MainTex",new Vector2(1.00f/o_x,1.000f/o_y));
        StartCoroutine(I_gg());
 }
 void Update () {
 }
    Vector2 vv;
    IEnumerator I_gg() {
        for(int i=0;i<o_y;i++) {
            for(int n=0;n<o_x;n++) {
                vv=new Vector2((n+0.0f)/o_x,(i+0.0f)/o_y);
                o_mm.material.SetTextureOffset("_MainTex",vv);
                yield return new WaitForSeconds(0.02f);
            }
        }
        StartCoroutine(I_gg()); 
    }
}
