#import <Foundation/Foundation.h>//命名空间
@interface C_dd : NSObject {  //--------------类的声明---------------------------
@public
    char *_name;
    int _age;
}
-(void) S_002;//无参数方法
-(void) S_003:(char *)ff;//1----
-(void) S_004:(char*)ff using:(char *)ff2;//2-----ß
-(void) S_005:(char*)ff :(char*)ff2 :(char*)ff3 :(char*)ff4;//4-------

@end//=================================================

@implementation C_dd//------------------类的实现------------------------------
-(void) S_002{//无参数方法的实现------
    NSLog(@"哈哈____ %s",_name);
}
-(void) S_003:(char *)ff{ //1-------
    NSLog(@"哈哈  吃了%s:",ff);
}
-(void) S_004:(char*)ff using:(char *)ff2{//2--------
     NSLog(@"哈哈  吃了%s: %s",ff,ff2);
}
-(void) S_005:(char*)ff :(char*)ff2 :(char*)ff3 :(char*)ff4{//4-------
      NSLog(@"哈哈  吃了%s: %s  %s  %s",ff,ff2,ff3,ff4);
}
@end//==================================================


int main(int argc, const char * argv[]) {
    @autoreleasepool {
       
        C_dd *cc =  [C_dd new];//类的创建
        cc->_name ="ddddaaaa李快快";  //负值
        cc->_age=20; //负值
        [cc S_002]; //调用无参数方法
        [cc S_003:"sssa"];
        [cc S_004:"sssa" using:"ddda"];
        [cc S_005:"sssa":"ddda":"uuua":"fffa"];
       
        NSLog(@"%p, %d = %d ",cc,cc->_age,(*cc)._age);
       
    }
    return 0;
}
