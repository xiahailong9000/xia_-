using UnityEngine;
using System.Collections;
public class kao_bian1 : MonoBehaviour{
    public float y0 =0;
    public float y1 =300;
    public float x0 = 300;
    public float x1 = 200;
    bool bb = false ;
    Vector2 cc = new Vector2(44, 66);
    int ii1 =1;  //面板切换
    string ss = "aaaaaa";
    void Start(){
    }
    void Update(){
        if ((Input.mousePosition.x > x0) && (Input.mousePosition.x < x0 + x1)){   //鼠标在面板区域  面板伸出
            if ((Screen.height - Input.mousePosition.y < y0 + y1) && (y0 < 0))       y0 += 10.0f;
        }
        if(!bb)  //鼠标在面板外部  面板退回
        if(( ((Input.mousePosition.x < x0) || (Input.mousePosition.x > x0 + x1))||(Screen.height - Input.mousePosition.y > y0 + y1)) && (y0 > -1 * y1 + 25))           y0 -= 10.0f;
    }
    void OnGUI(){
        GUI.BeginGroup(new Rect(x0, y0, x1, y1));

        if (GUI.Button(new Rect(0, 0, x1 / 3, 20), "aaaaaa")) { ii1 = 1; ss = "aaaaaaa "; }
        if (GUI.Button(new Rect(x1 / 3, 0, x1 / 3, 20), "sssssss ")) { ii1 = 2; ss = "sssssss "; }
        if (GUI.Button(new Rect(x1 * 2 / 3, 0, x1 / 3, 20), "fffffff")) { ii1 = 3; ss = "ffffffff "; }

        if (ii1 == 1){
            GUILayout.BeginArea(new Rect(0, 23, x1, y1 - 46));//面板集开始 ============================
            this.cc = GUILayout.BeginScrollView(this.cc);   //自适应滚动条面板
            GUILayout.Button("聊天室3");
            GUILayout.Button("聊天室3");
            GUILayout.Button("聊天室3");
            GUILayout.Button("聊天室3");
            GUILayout.Button("聊天室3");
            GUILayout.Button("聊天室3");
            GUILayout.Button("聊天室3");
            GUILayout.Button("聊天室2");
            GUILayout.Button("聊天室3");
            GUILayout.Button("聊天室3");
            GUILayout.Button("聊天室3");
            GUILayout.Button("聊天室3");
            GUILayout.Button("聊天室3");
            GUILayout.Button("聊天室3");
            GUILayout.Button("聊天室3");
            GUILayout.EndScrollView();//从下面排列
            GUILayout.EndArea();//面板集结束===========================================
        }
        if (ii1 == 2) GUI.Box(new Rect(0, 23, x1, y1 - 46), "");
        if (ii1 == 3) GUI.TextArea(new Rect(0, 23, x1, y1 - 46), "dfdfdfd\nfdfdfd\ndfdfd\n\n\n\n\n\n\n\n\n\n\n\n\n\nn\fgdfgdf");

        GUI.Box(new Rect(0, y1 - 20, x1, 20), "");
        if (GUI.RepeatButton(new Rect(0, y1 - 20, x1 - 23, 20), ss)) x0 = x0 + Input.GetAxis("Mouse X")*15;//鼠标推动底部左右移动


        bb = GUI.Toggle(new Rect(x1 - 15, y1 - 18, 18, 20), !bb, ""); //定点开关
        GUI.EndGroup();//组结束
    }
}
