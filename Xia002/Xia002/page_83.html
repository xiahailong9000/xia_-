using UnityEngine;
using System.Collections;
public class C_ll : MonoBehaviour {
    GameObject o_平移辅助物体,o_相机父物体,o_拉伸辅助物体;
    Vector3 v平移判断,v平移前置,v拉伸前置;
  float o_最左=-300,o_最右=300,o_最前=300,o_最后=-300,o_最上=300,o_最下=10;
 public bool o_缓冲开关;
 void Start () {
  Vector3 vv=transform.position;
  transform.position=new Vector3(vv.x,20,vv.z);
  vv=transform.position;
  o_平移辅助物体=new GameObject("Camera平移辅助物体");
  o_平移辅助物体.transform.position=vv;
  o_平移辅助物体.transform.localEulerAngles=new Vector3(0,transform.localEulerAngles.y,0);
  o_相机父物体=new GameObject("Camera相机父物体");
  o_相机父物体.transform.position=vv;
  o_相机父物体.transform.localEulerAngles=new Vector3(0,transform.localEulerAngles.y,0);
  o_拉伸辅助物体=new GameObject("Camera拉伸辅助物体");
  o_拉伸辅助物体.transform.position=vv;
  o_拉伸辅助物体.transform.localEulerAngles=transform.localEulerAngles;
  transform.SetParent(o_相机父物体.transform);
  o_拉伸辅助物体.transform.SetParent(o_相机父物体.transform);
 }
 void Update () {
  if(Input.GetMouseButton(0)) {
   v平移前置=o_平移辅助物体.transform.InverseTransformDirection(o_平移辅助物体.transform.position);
   o_平移辅助物体.transform.Translate(Vector3.right*-Input.GetAxis("Mouse X")*o_拉伸辅助物体.transform.position.y*0.5f);
   o_平移辅助物体.transform.Translate(Vector3.forward*-Input.GetAxis("Mouse Y")*o_拉伸辅助物体.transform.position.y*0.5f);
  }
  v拉伸前置=o_拉伸辅助物体.transform.position;
  o_拉伸辅助物体.transform.Translate(Vector3.forward*Input.GetAxis("Mouse ScrollWheel")*o_拉伸辅助物体.transform.position.y);
     v平移判断=o_平移辅助物体.transform.InverseTransformDirection(o_平移辅助物体.transform.position);
  if((v平移判断.x>o_最右||v平移判断.x<o_最左)||(v平移判断.z>o_最前||v平移判断.z<o_最后)) {
   v平移判断=v平移前置;
  }
  o_平移辅助物体.transform.position=o_平移辅助物体.transform.TransformDirection(v平移判断);
  if(o_拉伸辅助物体.transform.position.y>o_最上||o_拉伸辅助物体.transform.position.y<o_最下) {
   o_拉伸辅助物体.transform.position=v拉伸前置;
  }
  if(o_缓冲开关) {
   transform.position=Vector3.Lerp(transform.position,o_拉伸辅助物体.transform.position,0.05f);
   o_相机父物体.transform.position=Vector3.Lerp(o_相机父物体.transform.position,o_平移辅助物体.transform.position,0.05f);
  } else {
   transform.position=o_拉伸辅助物体.transform.position;
   o_相机父物体.transform.position=o_平移辅助物体.transform.position;
  }
 }
}

 

 

 

 

//Android--IOS平台下不显示阴影， -------------------------------------------------------------
1.把Edit-GraphicsEmulation-NoEmulation勾上，
2.出apk包时最好能把playersetting-GraphicsLevel改为force OpenGL ES 3.0 ，不改部分手机也能有阴影
3.qualitysetting-shadowdistance要合适
4.camera离阴影的物理距离不能太远
5.fast、fastest下没有阴影
6.光源要hardshadow或者softshadow，有些硬件不支持softshadow
7.投射物体材质球要cast shadow，接收物体要receiveshadow勾上
