//计时器封装-------------： C_计时器.S_计时(1000, S_002);

public class C_计时器{
    public delegate void dell();
    static System.Timers.Timer t;
    static dell del;
    public static void S_计时(int 时间, dell de){
        t = new System.Timers.Timer(时间*1000);
        t.Elapsed += S_002;
        del = de;
        t.AutoReset = false;
        t.Enabled = true;
    }
    static void S_002(object source, ElapsedEventArgs e){
        del.Invoke();
    }
}
