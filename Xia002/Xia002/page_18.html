using System;

using UnityEngine;

using System.Collections;

public class TimePassword : MonoBehaviour{
    private int timepassword;


    void Start(){
        timepassword = int.Parse(System.DateTime.Now.ToString("yyyyMMddHHmmss").Substring(0,8));
        if (timepassword > 20141205)
            Application.Quit();
 
    }

}
