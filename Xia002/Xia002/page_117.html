        static string S_加密2(string ss) {
            byte[] zData=Encoding.UTF8.GetBytes(ss);
            List<byte> z列表=new List<byte>(zData);
            for(int i=0;i<z列表.Count;i++) {
                //Console.WriteLine(i+"_________"+z列表[i]);
                bool z循环开关=false;
                if(o_开始符.Contains(z列表[i])==true) {
                    if(z列表.Count>i+1&&z列表[i+1]==95) {//"_"  
                        z循环开关=true;
                    }
                }
                if(z循环开关==false) {
                    if(o_开始符2.Contains(z列表[i])==true) {
                        z循环开关=true;
                    }
                }
                while(z循环开关==true) {                  
                    if(z列表.Count-1<i) {                    
                        break;
                    }          
                    if(z列表[i]>127) {
                        string sa=z列表[i].ToString();
                        int i01=0,i02=0;
                        if(z列表[i]<200) {
                            i01=int.Parse(sa.Substring(1,1))+101;
                            i02=int.Parse(sa.Substring(2,1))+102;
                        } else {
                            i01=int.Parse(sa.Substring(1,1))+66;
                            i02=int.Parse(sa.Substring(2,1))+68;
                        }
                        z列表[i]=(byte)i01;
                        z列表.Insert(i+1,(byte)i02);                     
                    }
                    i++;
                    if(i<z列表.Count&&z列表[i]<127) {//结束符号---
                        if(o_结束符.Contains(z列表[i])) {
                            z循环开关=false;
                            break;
                        }
                    }
                }
            }
            zData=z列表.ToArray();
            ss=Encoding.UTF8.GetString(zData,0,zData.Length);
            return ss;
        }       
        public static List<string> S_获取目录下_所有文件(string path,string type) {
            List<string> list = new List<string>();
            //遍历文件夹
            DirectoryInfo theFolder = new DirectoryInfo(path);
            FileInfo[] z子文件集 = theFolder.GetFiles(type,SearchOption.AllDirectories);
            for(int i=0;i<z子文件集.Length;i++) {
                list.Add(z子文件集[i].FullName);
            }
            return list;
        }
        public static List<string> S_Txt读取(string path) {
            StreamReader sr = new StreamReader(path,Encoding.UTF8);
            String line;
            List<string> z列表 = new List<string>();
            while((line=sr.ReadLine())!=null) {
                z列表.Add(line);
            }
            sr.Close();
            return z列表;
        }
        static void S_Txt写入(List<string> z内容列表,string z路径) {
            if(File.Exists(z路径))
                File.Delete(z路径); //文件存在就删除
            FileInfo fi = new FileInfo(z路径);
            var di = fi.Directory;
            if(!di.Exists)//判断目录和文件是否存在  不存在就创建---
                di.Create();
            StreamWriter sw=File.AppendText(z路径);
            for(int i = 0;i<z内容列表.Count;i++) {
                sw.WriteLine(z内容列表[i]);
            }
            sw.Close();
        }
